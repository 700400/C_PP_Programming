``` c++
#include<iostream>
using namespace std;
class CPnt{
	private: int w,h;
	public:
		CPnt(int a, int b){
			w = a , h = b;
		}
		CPnt(){
			w=0,h=0;
		}
		void Pr(){
			cout << w << "," << h << endl;
		}
		void Prt(){
			cout << w*h << endl;
		}
		CPnt Add(CPnt t){
			CPnt r;
			r.w = w + t.w;
			r.h = h + t.h;
			return r;
		}
};
int main(){
	CPnt p1(2,2), p2(3,3), p3;
	p3 = p1.Add(p2);
	p3.Pr();
	p3.Prt();
	return 0;
}
```
